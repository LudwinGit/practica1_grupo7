public Boolean validarPrimo(int numero)
{
    if(numero < 0)
        return false;
        
    int contador = 0;
    for (int i = 1; i <= numero ; i++)
    {
        if (numero % i == 0)
            contador++;
    }
    
    if (contador == 2)
        return true;
    else
        return false;
}
